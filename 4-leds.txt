const int leds[] = {6, 7, 8, 9};   // pinos dos LEDs
const int numLeds = 4;
const int buzzerPin = 10;          // pino do buzzer

// notas diferentes (em Hz) para cada LED
const int tones[] = {523, 659, 784, 988};  
// 523 = D칩, 659 = Mi, 784 = Sol, 988 = Si

int baseDelay = 300;    // tempo inicial entre LEDs (ms)
int minDelay = 80;      // velocidade m치xima (limite m칤nimo)
int step = 2;           // quanto reduz por piscada (menor = mais suave)

void setup() {
  for (int i = 0; i < numLeds; i++) {
    pinMode(leds[i], OUTPUT);
  }
  pinMode(buzzerPin, OUTPUT);

  // Inicializa o gerador de n칰meros aleat칩rios
  randomSeed(analogRead(A0));  // use um pino anal칩gico n칚o conectado
}

void loop() {
  int ledIndex = random(numLeds);   // escolhe LED aleat칩rio

  digitalWrite(leds[ledIndex], HIGH);
  tone(buzzerPin, tones[ledIndex]);
  delay(baseDelay / 2);
  noTone(buzzerPin);

  delay(baseDelay / 2);
  digitalWrite(leds[ledIndex], LOW);

  // 游대 Acelera a cada piscada
  if (baseDelay > minDelay) {
    baseDelay -= step;
  } 
  else {
    // Reinicia a velocidade quando atingir o m치ximo
    baseDelay = 300;
  }
}
